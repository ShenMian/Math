29 inline void matrixMul(const __m128 m1[4], const __m128 m2[4], __m128 dst[4])
30 {
31  __m128 tmp[4];
32  {
33  __m128 e0 = _mm_shuffle_ps(m2[0], m2[0], _MM_SHUFFLE(0, 0, 0, 0));
34  __m128 e1 = _mm_shuffle_ps(m2[0], m2[0], _MM_SHUFFLE(1, 1, 1, 1));
35  __m128 e2 = _mm_shuffle_ps(m2[0], m2[0], _MM_SHUFFLE(2, 2, 2, 2));
36  __m128 e3 = _mm_shuffle_ps(m2[0], m2[0], _MM_SHUFFLE(3, 3, 3, 3));
37 
38  __m128 v0 = _mm_mul_ps(m1[0], e0);
39  __m128 v1 = _mm_mul_ps(m1[1], e1);
40  __m128 v2 = _mm_mul_ps(m1[2], e2);
41  __m128 v3 = _mm_mul_ps(m1[3], e3);
42 
43  __m128 a0 = _mm_add_ps(v0, v1);
44  __m128 a1 = _mm_add_ps(v2, v3);
45  __m128 a2 = _mm_add_ps(a0, a1);
46 
47  tmp[0] = a2;
48  }
49 
50  {
51  __m128 e0 = _mm_shuffle_ps(m2[1], m2[1], _MM_SHUFFLE(0, 0, 0, 0));
52  __m128 e1 = _mm_shuffle_ps(m2[1], m2[1], _MM_SHUFFLE(1, 1, 1, 1));
53  __m128 e2 = _mm_shuffle_ps(m2[1], m2[1], _MM_SHUFFLE(2, 2, 2, 2));
54  __m128 e3 = _mm_shuffle_ps(m2[1], m2[1], _MM_SHUFFLE(3, 3, 3, 3));
55 
56  __m128 v0 = _mm_mul_ps(m1[0], e0);
57  __m128 v1 = _mm_mul_ps(m1[1], e1);
58  __m128 v2 = _mm_mul_ps(m1[2], e2);
59  __m128 v3 = _mm_mul_ps(m1[3], e3);
60 
61  __m128 a0 = _mm_add_ps(v0, v1);
62  __m128 a1 = _mm_add_ps(v2, v3);
63  __m128 a2 = _mm_add_ps(a0, a1);
64 
65  tmp[1] = a2;
66  }
67 
68  {
69  __m128 e0 = _mm_shuffle_ps(m2[2], m2[2], _MM_SHUFFLE(0, 0, 0, 0));

71  __m128 e2 = _mm_shuffle_ps(m2[2], m2[2], _MM_SHUFFLE(2, 2, 2, 2));
72  __m128 e3 = _mm_shuffle_ps(m2[2], m2[2], _MM_SHUFFLE(3, 3, 3, 3));
73 
74  __m128 v0 = _mm_mul_ps(m1[0], e0);
75  __m128 v1 = _mm_mul_ps(m1[1], e1);
76  __m128 v2 = _mm_mul_ps(m1[2], e2);
77  __m128 v3 = _mm_mul_ps(m1[3], e3);
78 
79  __m128 a0 = _mm_add_ps(v0, v1);
80  __m128 a1 = _mm_add_ps(v2, v3);
81  __m128 a2 = _mm_add_ps(a0, a1);
82 
83  tmp[2] = a2;
84  }
85 
86  {
87  __m128 e0 = _mm_shuffle_ps(m2[3], m2[3], _MM_SHUFFLE(0, 0, 0, 0));
88  __m128 e1 = _mm_shuffle_ps(m2[3], m2[3], _MM_SHUFFLE(1, 1, 1, 1));
89  __m128 e2 = _mm_shuffle_ps(m2[3], m2[3], _MM_SHUFFLE(2, 2, 2, 2));

119 inline void matrixTranspose(const __m128 m[4], __m128 dst[4])
120 {
121  __m128 tmp0 = _mm_shuffle_ps(m[0], m[1], 0x44);
122  __m128 tmp2 = _mm_shuffle_ps(m[0], m[1], 0xEE);
123  __m128 tmp1 = _mm_shuffle_ps(m[2], m[3], 0x44);
124  __m128 tmp3 = _mm_shuffle_ps(m[2], m[3], 0xEE);
125 
126  dst[0] = _mm_shuffle_ps(tmp0, tmp1, 0x88);
127  dst[1] = _mm_shuffle_ps(tmp0, tmp1, 0xDD);
128  dst[2] = _mm_shuffle_ps(tmp2, tmp3, 0x88);
129  dst[3] = _mm_shuffle_ps(tmp2, tmp3, 0xDD);
130 }

132 inline void matrixMulVec(const __m128 m[4], const __m128& v, __m128& dst)
133 {
134  __m128 col1 = _mm_shuffle_ps(v, v, _MM_SHUFFLE(0, 0, 0, 0));
135  __m128 col2 = _mm_shuffle_ps(v, v, _MM_SHUFFLE(1, 1, 1, 1));
136  __m128 col3 = _mm_shuffle_ps(v, v, _MM_SHUFFLE(2, 2, 2, 2));
137  __m128 col4 = _mm_shuffle_ps(v, v, _MM_SHUFFLE(3, 3, 3, 3));
138 
139  dst = _mm_add_ps(
140  _mm_add_ps(_mm_mul_ps(m[0], col1), _mm_mul_ps(m[1], col2)),
141  _mm_add_ps(_mm_mul_ps(m[2], col3), _mm_mul_ps(m[3], col4))
142  );
143 }
